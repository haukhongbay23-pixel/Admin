-- 1. HÀM HỖ TRỢ (NOCLIP & ANTI-AFK)
local function Noclip()
    pcall(function()
        local char = game.Players.LocalPlayer.Character
        for _, v in pairs(char:GetDescendants()) do
            if v:IsA("BasePart") and v.CanCollide == true then
                v.CanCollide = false
            end
        end
    end)
end

local function CreateToggle(name, default, callback)
    local Button = Instance.new("TextButton")
    Button.Size = UDim2.new(1, -10, 0, 35)
    Button.BackgroundColor3 = default and Color3.fromRGB(180, 0, 0) or Color3.fromRGB(40, 40, 40)
    Button.Text = name .. (default and ": ON" or ": OFF")
    Button.TextColor3 = Color3.new(1, 1, 1)
    Button.Font = Enum.Font.GothamSemibold
    Button.Parent = ContentFrame
    
    local Corner = Instance.new("UICorner")
    Corner.CornerRadius = UDim.new(0, 6)
    Corner.Parent = Button

    local state = default
    Button.MouseButton1Click:Connect(function()
        state = not state
        Button.Text = name .. (state and ": ON" or ": OFF")
        Button.BackgroundColor3 = state and Color3.fromRGB(180, 0, 0) or Color3.fromRGB(40, 40, 40)
        callback(state)
    end)
end

-- 3. LOGIC GAME
function CheckHasQuest()
    local pGui = game.Players.LocalPlayer:FindFirstChild("PlayerGui")
    if pGui and pGui:FindFirstChild("Main") and pGui.Main:FindFirstChild("Quest") then
        return pGui.Main.Quest.Visible
    end
    return false
end

-- Tự động tăng điểm (Stats)
spawn(function()
    while task.wait(1) do
        if _G.AutoStats then
            game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("AddPoint", "Melee", 3)
        end
    end
end)

-- Vòng lặp Gom quái & Đánh
spawn(function()
    while task.wait() do
        if _G.AutoFarm then
            pcall(function()
                local char = game.Players.LocalPlayer.Character
                if _G.BringMobs then
                    for _, v in pairs(game:GetService("Workspace").Enemies:GetChildren()) do
                        if v.Name == MonsterName and v:FindFirstChild("HumanoidRootPart") and v.Humanoid.Health > 0 then
                            v.HumanoidRootPart.CanCollide = false
                            v.HumanoidRootPart.CFrame = char.HumanoidRootPart.CFrame * CFrame.new(0, 0, -5)
                        end
                    end
                end
                if _G.FastAttack and char:FindFirstChildOfClass("Tool") then
                    vu:CaptureController()
                    vu:Button1Down(Vector2.new(1280, 672))
                end
            end)
        end
    end
end)

-- Vòng lặp Farm chính
spawn(function()
    while task.wait(0.1) do
        if _G.AutoFarm then
            pcall(function()
                local player = game.Players.LocalPlayer
                local char = player.Character
                local root = char.HumanoidRootPart
                
                Noclip() -- Luôn giữ trạng thái xuyên địa hình khi farm

                if _G.AutoBuso and not char:FindFirstChild("HasBuso") then
                    game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("Buso")
                end
                
                if not char:FindFirstChildOfClass("Tool") then
                    local w = player.Backpack:FindFirstChild(_G.SelectWeapon)
                    if w then char.Humanoid:EquipTool(w) end
                end

                if not CheckHasQuest() then
                    local npc = game:GetService("Workspace").NPCs:FindFirstChild(QuestNPC)
                    if npc then
                        root.CFrame = npc.HumanoidRootPart.CFrame * CFrame.new(0, 5, 0)
                        task.wait(0.2)
                        game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("StartQuest", QuestName, 1)
                    end
                else
                    local enemy = game:GetService("Workspace").Enemies:FindFirstChild(MonsterName)
                    if enemy and enemy:FindFirstChild("HumanoidRootPart") then
                        root.CFrame = enemy.HumanoidRootPart.CFrame * CFrame.new(0, 8, 0) -- Bay trên đầu quái
                    else
                        local sp = game:GetService("Workspace").EnemySpawns:FindFirstChild(MonsterName)
                        if sp then root.CFrame = sp.CFrame * CFrame.new(0, 20, 0) end
                    end
                end
            end)
        end
    end
end)
