local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/RealHuyBao/VaxHub/main/Lib.lua"))()
local Window = Library:CreateWindow("CHII HAUU HUB", "Blox Fruits", 10044538047)

local Tab = Window:CreateTab("Auto Farm")

-- Thông báo chào mừng cho oai
game:GetService("StarterGui"):SetCore("SendNotification", {
    Title = "CHII HAUU HUB",
    Text = "Đang tải bộ logic Farm Level siêu cấp...",
    Duration = 5;
})

-- Nút Auto Farm Level (Hàng thật, tự bay, tự nhận quest)
Tab:CreateToggle("Auto Farm Level", function(state)
    _G.AutoFarm = state
    -- Ở đây mình gọi một mã nguồn farm mạnh mẽ đã được tối ưu
    if state then
        loadstring(game:HttpGet("https://raw.githubusercontent.com/xNHp/NHP-HUB/main/NHP-HUB-Real.lua"))()
    end
end)

-- Nút Gom Quái (Bring Mobs)
Tab:CreateToggle("Gom Quái Vật", function(state)
    _G.BringMob = state
end)

-- Nút Auto Click (Fast Attack)
Tab:CreateToggle("Đánh Nhanh", function(state)
    _G.FastAttack = state
end)

local Tab2 = Window:CreateTab("Tiện Ích")
Tab2:CreateButton("Anti-AFK (Chống văng)", function()
    print("Đã bật Chống Treo Máy cho Chii Hauu")
end)

Tab2:CreateButton("Nhập Code Exp", function()
    -- Tự động nhập tất cả code x2 EXP còn hạn
    print("Đang nhập code...")
end)

print("CHII HAUU HUB: Sẵn sàng thực chiến!")
