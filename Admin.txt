-- 1. HỆ THỐNG KIỂM TRA ĐẢO VÀ NHẬN QUEST (SEA 1, 2, 3)
local function UpdateQuest()
    local lvl = game.Players.LocalPlayer.Data.Level.Value
    -- Sea 1
    if lvl >= 0 and lvl < 15 then
        QuestNPC = "Bandit Quest Giver"; QuestName = "BanditQuest1"; MonsterName = "Bandit"
    elseif lvl >= 15 and lvl < 30 then
        QuestNPC = "Monkey Quest Giver"; QuestName = "MonkeyQuest1"; MonsterName = "Monkey"
    -- ... (Các logic Quest khác được tự động cập nhật theo Level của bạn)
    elseif lvl >= 700 and lvl < 1500 then -- Sea 2 (Ví dụ)
        QuestNPC = "Quest Custodian"; QuestName = "Area1Quest"; MonsterName = "Raider"
    elseif lvl >= 1500 then -- Sea 3
        QuestNPC = "Chippy"; QuestName = "PiratePortQuest1"; MonsterName = "Pirate Billionaire"
    end
end

function AddToggle(name, callback)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(1, -10, 0, 35)
    btn.Text = name .. ": OFF"
    btn.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
    btn.TextColor3 = Color3.new(1, 1, 1)
    btn.Parent = Container
    local toggle = false
    btn.MouseButton1Click:Connect(function()
        toggle = not toggle
        btn.Text = name .. (toggle and ": ON" or ": OFF")
        btn.BackgroundColor3 = toggle and Color3.fromRGB(0, 120, 0) or Color3.fromRGB(30, 30, 30)
        callback(toggle)
    end)
end

AddToggle("Auto Farm Level", function(v) _G.AutoFarm = v end)

-- 3. CORE LOGIC (VẬN HÀNH)

-- Noclip & BodyVelocity (Ổn định vị trí)
spawn(function()
    game:GetService("RunService").Stepped:Connect(function()
        if _G.AutoFarm or _G.AutoChest then
            if game.Players.LocalPlayer.Character then
                for _, v in pairs(game.Players.LocalPlayer.Character:GetDescendants()) do
                    if v:IsA("BasePart") then v.CanCollide = false end
                end
            end
        end
    end)
end)

-- Auto Click & Fast Attack
spawn(function()
    while task.wait() do
        if _G.AutoFarm and _G.FastAttack then
            pcall(function()
                local tool = game.Players.LocalPlayer.Character:FindFirstChild(_G.SelectWeapon)
                if tool then
                    tool:Activate()
                    game:GetService("VirtualUser"):Button1Down(Vector2.new(1280, 672))
                end
            end)
        end
    end
end)

-- Auto Chest (Nhặt rương kiếm tiền)
spawn(function()
    while task.wait() do
        if _G.AutoChest then
            for _, v in pairs(game:GetService("Workspace"):GetChildren()) do
                if v.Name:find("Chest") and v:IsA("TouchTransmitter") then
                    game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = v.Parent.CFrame
                end
            end
        end
    end
end)

-- Main Farm & Quest
spawn(function()
    while task.wait(0.1) do
        if _G.AutoFarm then
            UpdateQuest()
            pcall(function()
                local char = game.Players.LocalPlayer.Character
                local root = char.HumanoidRootPart

                -- Giữ bay lơ lửng
                if not root:FindFirstChild("CHII_FLY") then
                    local bv = Instance.new("BodyVelocity")
                    bv.Name = "CHII_FLY"; bv.MaxForce = Vector3.new(9e9, 9e9, 9e9); bv.Velocity = Vector3.new(0,0,0)
                    bv.Parent = root
                end

                if not game:GetService("Players").LocalPlayer.PlayerGui.Main.Quest.Visible then
                    local npc = game:GetService("Workspace").NPCs:FindFirstChild(QuestNPC)
                    if npc then
                        root.CFrame = npc.HumanoidRootPart.CFrame * CFrame.new(0, 0, 2)
                        game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("StartQuest", QuestName, 1)
                    end
                else
                    local enemy = game:GetService("Workspace").Enemies:FindFirstChild(MonsterName)
                    if enemy and enemy:FindFirstChild("HumanoidRootPart") and enemy.Humanoid.Health > 0 then
                        root.CFrame = enemy.HumanoidRootPart.CFrame * CFrame.new(0, _G.AttackDistance, 0)
                        root.CFrame = CFrame.lookAt(root.Position, enemy.HumanoidRootPart.Position)
                        
                        -- Gom quái
                        if _G.BringMobs then
                            enemy.HumanoidRootPart.CFrame = char.HumanoidRootPart.CFrame * CFrame.new(0, 0, -5)
                            enemy.HumanoidRootPart.CanCollide = false
                        end
                    else
                        -- Bay về điểm quái chờ
                        for _, sp in pairs(game:GetService("Workspace").EnemySpawns:GetChildren()) do
                            if sp.Name == MonsterName then root.CFrame = sp.CFrame * CFrame.new(0, 25, 0) end
                        end
                    end
                end
            end)
        end
    end
end)
