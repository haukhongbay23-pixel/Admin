repeat task.wait() until game:IsLoaded()

-- Cài đặt
_G.AutoFarm = false
_G.BringMobs = false
_G.FastAttack = true 
_G.AttackDistance = 8 
_G.AutoStats = false 
_G.SelectWeapon = "" 

local QuestNPC, QuestName, MonsterName = "", "", ""

-- ANTI-AFK

    vu:Button2Down(Vector2.new(0,0),workspace.CurrentCamera.CFrame)

-- 1. UPDATE QUEST LOGIC (SEA 1 FULL)

    if lvl >= 0 and lvl < 15 then

        QuestNPC = "Monkey Quest Giver"; QuestName = "MonkeyQuest1"; MonsterName = "Monkey"
    elseif lvl >= 30 and lvl < 60 then
        QuestNPC = "Pirate Adventurer"; QuestName = "PirateQuest1"; MonsterName = "Pirate"
    elseif lvl >= 60 and lvl < 90 then
        QuestNPC = "Desert Adventurer"; QuestName = "DesertQuest"; MonsterName = "Desert Bandit"
    elseif lvl >= 90 and lvl < 120 then
        QuestNPC = "Snow Quest Giver"; QuestName = "SnowQuest"; MonsterName = "Snow Bandit"
    elseif lvl >= 120 and lvl < 150 then
        QuestNPC = "Chief Petty Officer"; QuestName = "MarineQuest2"; MonsterName = "Chief Petty Officer"
    elseif lvl >= 150 and lvl < 190 then
        QuestNPC = "Sky Quest Giver"; QuestName = "SkyQuest"; MonsterName = "Sky Bandit"
    elseif lvl >= 190 and lvl < 250 then
        QuestNPC = "Prisoner Quest Giver"; QuestName = "PrisonerQuest"; MonsterName = "Prisoner"
    elseif lvl >= 250 and lvl < 300 then
        QuestNPC = "Magma Quest Giver"; QuestName = "MagmaQuest"; MonsterName = "Military Soldier"
    elseif lvl >= 300 and lvl < 330 then
        QuestNPC = "Magma Quest Giver"; QuestName = "MagmaQuest"; MonsterName = "Military Spy"
    elseif lvl >= 330 and lvl < 375 then
        QuestNPC = "Fishman Quest Giver"; QuestName = "FishmanQuest"; MonsterName = "Fishman Warrior"
    elseif lvl >= 375 and lvl < 400 then
        QuestNPC = "Fishman Quest Giver"; QuestName = "FishmanQuest"; MonsterName = "Fishman Commando"
    elseif lvl >= 400 and lvl < 525 then
        QuestNPC = "Wysper"; QuestName = "SkyExp1Quest"; MonsterName = "God's Guard"
    elseif lvl >= 525 and lvl < 625 then
        QuestNPC = "Wysper"; QuestName = "SkyExp2Quest"; MonsterName = "Shanda"
    elseif lvl >= 625 and lvl < 700 then
        QuestNPC = "Cyborg Quest Giver"; QuestName = "LastVillageQuest"; MonsterName = "Galley Pirate"
    elseif lvl >= 700 then
        QuestNPC = "Quest Custodian"; QuestName = "Area1Quest"; MonsterName = "Raider"
    end
end

-- 2. SIÊU FAST ATTACK (Sử dụng Tool Click liên tục)
spawn(function()
    while task.wait() do

local WeaponBtn = Instance.new("TextButton")

WeaponBtn.Text = "👉 CHỌN VŨ KHÍ TRÊN TAY"
WeaponBtn.Parent = Content
WeaponBtn.MouseButton1Click:Connect(function()
    local tool = game.Players.LocalPlayer.Character:FindFirstChildOfClass("Tool")
    if tool then 
        _G.SelectWeapon = tool.Name 
        WeaponBtn.Text = "Đã chọn: "..tool.Name
    end
end)

function CreateBtn(name, callback)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(1, -10, 0, 40)
    btn.Text = name .. ": OFF"
    btn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
    btn.TextColor3 = Color3.new(1, 1, 1)
    btn.Parent = Content
    local state = false
    btn.MouseButton1Click:Connect(function()
        state = not state
        btn.Text = name .. (state and ": ON" or ": OFF")
        btn.BackgroundColor3 = state and Color3.fromRGB(0, 180, 0) or Color3.fromRGB(45, 45, 45)
        callback(state)
    end)
end

CreateBtn("Auto Farm Level", function(v) _G.AutoFarm = v end)
CreateBtn("Gom Quái (Bring)", function(v) _G.BringMobs = v end)
CreateBtn("Siêu Đánh Nhanh", function(v) _G.FastAttack = v end)
CreateBtn("Auto Stats Melee", function(v) _G.AutoStats = v end)

-- 4. LOGIC VẬT LÝ (NOCLIP)
spawn(function()
    game:GetService("RunService").Stepped:Connect(function()
        if _G.AutoFarm then
            if game.Players.LocalPlayer.Character then
                for _, v in pairs(game.Players.LocalPlayer.Character:GetDescendants()) do
                    if v:IsA("BasePart") then v.CanCollide = false end
                end
            end
        end
    end)
end)

-- 5. AUTO STATS
spawn(function()
    while task.wait(0.5) do
        if _G.AutoStats then
            game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("AddPoint", "Melee", 1)
        end
    end
end)

-- 6. FARM LOGIC (CHÍNH)
spawn(function()
    while task.wait(0.1) do
        if _G.AutoFarm then
            UpdateQuest()
            pcall(function()
                local char = game.Players.LocalPlayer.Character
                local root = char.HumanoidRootPart

                -- BodyVelocity giữ thăng bằng
                if not root:FindFirstChild("CHII_FLY") then
                    local bv = Instance.new("BodyVelocity")
                    bv.Name = "CHII_FLY"
                    bv.MaxForce = Vector3.new(9e9, 9e9, 9e9)
                    bv.Velocity = Vector3.new(0, 0, 0)
                    bv.Parent = root
                end

                -- Trang bị vũ khí
                if _G.SelectWeapon ~= "" and not char:FindFirstChild(_G.SelectWeapon) then
                    local tool = game.Players.LocalPlayer.Backpack:FindFirstChild(_G.SelectWeapon)
                    if tool then char.Humanoid:EquipTool(tool) end
                end

                if not game:GetService("Players").LocalPlayer.PlayerGui.Main.Quest.Visible then
                    -- Đi nhận Quest
                    local npc = game:GetService("Workspace").NPCs:FindFirstChild(QuestNPC)
                    if npc then
                        root.CFrame = npc.HumanoidRootPart.CFrame * CFrame.new(0, 0, 1)
                        task.wait(0.2)
                        game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("StartQuest", QuestName, 1)
                    end
                else
                    -- Đi đánh quái
                    local enemy = game:GetService("Workspace").Enemies:FindFirstChild(MonsterName)
                    if enemy and enemy:FindFirstChild("HumanoidRootPart") and enemy.Humanoid.Health > 0 then
                        root.CFrame = enemy.HumanoidRootPart.CFrame * CFrame.new(0, _G.AttackDistance, 0)
                        root.CFrame = CFrame.lookAt(root.Position, enemy.HumanoidRootPart.Position)
                        
                        -- Gom quái (nếu bật)
                        if _G.BringMobs then
                            enemy.HumanoidRootPart.CFrame = char.HumanoidRootPart.CFrame * CFrame.new(0, 0, -5)
                            enemy.HumanoidRootPart.CanCollide = false
                        end
                    else
                        -- Đợi quái hồi sinh
                        for _, sp in pairs(game:GetService("Workspace").EnemySpawns:GetChildren()) do
                            if sp.Name == MonsterName then
                                root.CFrame = sp.CFrame * CFrame.new(0, 30, 0)
                            end
                        end
                    end
                end
            end)
        else
            -- Xóa Fly khi tắt
            if game.Players.LocalPlayer.Character.HumanoidRootPart:FindFirstChild("CHII_FLY") then
                game.Players.LocalPlayer.Character.HumanoidRootPart.CHII_FLY:Destroy()
            end
        end
    end
end)
