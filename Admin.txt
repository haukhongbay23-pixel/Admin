-- 3. CHII HAUU - CORE LOGIC
local function GetNearestTarget()
    local nearest = nil
    local dist = 25 
    local myRoot = game.Players.LocalPlayer.Character and game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    
    if myRoot then
        for _, v in pairs(game:GetService("Workspace").Enemies:GetChildren()) do
            if v:FindFirstChild("HumanoidRootPart") and v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
                local d = (myRoot.Position - v.HumanoidRootPart.Position).Magnitude
                if d < dist then nearest = v; dist = d end
            end
        end
        for _, p in pairs(game:GetService("Players"):GetPlayers()) do
            if p ~= game.Players.LocalPlayer and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
                local d = (myRoot.Position - p.Character.HumanoidRootPart.Position).Magnitude
                if d < dist then nearest = p.Character; dist = d end
            end
        end
    end
    return nearest
end

-- THỰC THI AUTO STATS (MỚI BỔ SUNG)
spawn(function()
    while task.wait(1) do
        if _G.AutoStats then
            pcall(function()
                local points = game.Players.LocalPlayer.Data.StatsPoints.Value
                if points > 0 then
                    game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("AddPoint", "Melee", points)
                end
            end)
        end
    end
end)

spawn(function()
    while task.wait() do
        if _G.AutoFarm then
            pcall(function()
                local char = game.Players.LocalPlayer.Character
                if _G.SelectWeapon ~= "" and not char:FindFirstChild(_G.SelectWeapon) then
                    local tool = game.Players.LocalPlayer.Backpack:FindFirstChild(_G.SelectWeapon)
                    if tool then char.Humanoid:EquipTool(tool) end
                end
                
                if char:FindFirstChildOfClass("Tool") and _G.FastAttack then
                    if GetNearestTarget() then
                        game:GetService("VirtualUser"):CaptureController()
                        game:GetService("VirtualUser"):Button1Down(Vector2.new(1280, 672))
                    end
                end
                
                if _G.BringMobs then
                    for _, v in pairs(game:GetService("Workspace").Enemies:GetChildren()) do
                        if v.Name == MonsterName and v:FindFirstChild("HumanoidRootPart") and v.Humanoid.Health > 0 then
                            v.HumanoidRootPart.CanCollide = false
                            v.HumanoidRootPart.CFrame = char.HumanoidRootPart.CFrame * CFrame.new(0, 0, -5)
                        end
                    end
                end
            end)
        end
    end
end)

spawn(function()
    while task.wait(0.2) do
        if _G.AutoFarm then
            UpdateQuest() 
            pcall(function()
                local char = game.Players.LocalPlayer.Character
                if not char or not char:FindFirstChild("HumanoidRootPart") then return end
                for _, v in pairs(char:GetChildren()) do if v:IsA("BasePart") then v.CanCollide = false end end

                if not game:GetService("Players").LocalPlayer.PlayerGui.Main.Quest.Visible then
                    local npc = game:GetService("Workspace").NPCs:FindFirstChild(QuestNPC)
                    if npc then
                        char.HumanoidRootPart.CFrame = npc.HumanoidRootPart.CFrame * CFrame.new(0, 5, 0)
                        game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("StartQuest", QuestName, 1)
                    end
                else
                    local enemy = game:GetService("Workspace").Enemies:FindFirstChild(MonsterName)
                    if enemy and enemy:FindFirstChild("HumanoidRootPart") and enemy.Humanoid.Health > 0 then
                        char.HumanoidRootPart.CFrame = enemy.HumanoidRootPart.CFrame * CFrame.new(0, _G.AttackDistance, 0)
                    else
                        for _, sp in pairs(game:GetService("Workspace").EnemySpawns:GetChildren()) do
                            if sp.Name == MonsterName then
                                char.HumanoidRootPart.CFrame = sp.CFrame * CFrame.new(0, 20, 0)
                                break
                            end
                        end
                    end
                end
            end)
        end
    end
end)
