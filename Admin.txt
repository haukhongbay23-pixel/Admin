local function CreateToggle(name, default, callback)
    local ToggleFrame = Instance.new("Frame")
    ToggleFrame.Size = UDim2.new(1, -10, 0, 38)
    ToggleFrame.BackgroundTransparency = 1
    ToggleFrame.Parent = ContentFrame
    
    local Label = Instance.new("TextLabel")
    Label.Text = name
    Label.Size = UDim2.new(0.7, 0, 1, 0)
    Label.BackgroundTransparency = 1
    Label.TextColor3 = Color3.fromRGB(230, 230, 230)
    Label.Font = Enum.Font.GothamSemibold
    Label.TextXAlignment = Enum.TextXAlignment.Left
    Label.Parent = ToggleFrame

    local SlideBack = Instance.new("Frame")
    SlideBack.Size = UDim2.new(0, 45, 0, 22)
    SlideBack.Position = UDim2.new(1, -50, 0.5, -11)
    SlideBack.BackgroundColor3 = default and Color3.fromRGB(0, 180, 0) or Color3.fromRGB(60, 60, 60)
    SlideBack.Parent = ToggleFrame
    
    local SlideCorner = Instance.new("UICorner")
    SlideCorner.CornerRadius = UDim.new(1, 0)
    SlideCorner.Parent = SlideBack

    local Ball = Instance.new("Frame")
    Ball.Size = UDim2.new(0, 18, 0, 18)
    Ball.Position = default and UDim2.new(1, -20, 0.5, -9) or UDim2.new(0, 2, 0.5, -9)
    Ball.BackgroundColor3 = Color3.new(1, 1, 1)
    Ball.Parent = SlideBack

    local BallCorner = Instance.new("UICorner")
    BallCorner.CornerRadius = UDim.new(1, 0)
    BallCorner.Parent = Ball

    local state = default
    local Button = Instance.new("TextButton")
    Button.Size = UDim2.new(1, 0, 1, 0)
    Button.BackgroundTransparency = 1
    Button.Text = ""
    Button.Parent = SlideBack

    Button.MouseButton1Click:Connect(function()
        state = not state
        SlideBack.BackgroundColor3 = state and Color3.fromRGB(0, 180, 0) or Color3.fromRGB(60, 60, 60)
        Ball:TweenPosition(state and UDim2.new(1, -20, 0.5, -9) or UDim2.new(0, 2, 0.5, -9), "Out", "Quad", 0.2, true)
        callback(state)
    end)
end

-- 2. HỆ THỐNG LOGIC GAME
function CheckHasQuest()
    local pGui = game.Players.LocalPlayer:FindFirstChild("PlayerGui")
    if pGui and pGui:FindFirstChild("Main") and pGui.Main:FindFirstChild("Quest") then
        return pGui.Main.Quest.Visible
    end
    return false
end

-- Vòng lặp Gom & Đánh
spawn(function()
    while task.wait() do
        if _G.AutoFarm then
            pcall(function()
                local char = game.Players.LocalPlayer.Character
                if char and char:FindFirstChildOfClass("Tool") and _G.FastAttack then
                    game:GetService("VirtualUser"):Button1Down(Vector2.new(1280, 672))
                end

                if _G.BringMobs then
                    for _, v in pairs(game:GetService("Workspace").Enemies:GetChildren()) do
                        if v.Name == MonsterName and v:FindFirstChild("HumanoidRootPart") and v.Humanoid.Health > 0 then
                            if (v.HumanoidRootPart.Position - char.HumanoidRootPart.Position).Magnitude < 250 then
                                v.HumanoidRootPart.CanCollide = false
                                v.HumanoidRootPart.CFrame = char.HumanoidRootPart.CFrame * CFrame.new(0, 0, -5)
                            end
                        end
                    end
                end
            end)
        end
    end
end)

-- Vòng lặp Farm & Quest
spawn(function()
    while task.wait() do
        if _G.AutoFarm then
            pcall(function()
                local player = game.Players.LocalPlayer
                local char = player.Character
                if not char or not char:FindFirstChild("HumanoidRootPart") then return end
                
                -- Noclip chuyên nghiệp
                for _, v in pairs(char:GetDescendants()) do
                    if v:IsA("BasePart") then v.CanCollide = false end
                end

                if _G.AutoStats then game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("AddPoint", "Melee", 1) end
                
                if not char:FindFirstChildOfClass("Tool") then
                    local w = player.Backpack:FindFirstChild(_G.SelectWeapon)
                    if w then char.Humanoid:EquipTool(w) end
                end

                if not CheckHasQuest() then
                    local npc = game:GetService("Workspace").NPCs:FindFirstChild(QuestNPC)
                    if npc then
                        char.HumanoidRootPart.CFrame = npc.HumanoidRootPart.CFrame * CFrame.new(0, 5, 0)
                        task.wait(0.2)
                        game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("StartQuest", QuestName, 1)
                    end
                else
                    local enemy = game:GetService("Workspace").Enemies:FindFirstChild(MonsterName)
                    if enemy and enemy:FindFirstChild("HumanoidRootPart") and enemy.Humanoid.Health > 0 then
                        char.HumanoidRootPart.CFrame = enemy.HumanoidRootPart.CFrame * CFrame.new(0, _G.AttackDistance, 0)
                    else
                        local sp = game:GetService("Workspace").EnemySpawns:FindFirstChild(MonsterName)
                        if sp then char.HumanoidRootPart.CFrame = sp.CFrame * CFrame.new(0, 25, 0) end
                    end
                end
            end)
        end
    end
end)
