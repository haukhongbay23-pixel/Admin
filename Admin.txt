local function CreateToggle(name, default, callback)
    local Button = Instance.new("TextButton")
    Button.Size = UDim2.new(1, -10, 0, 35)
    Button.Text = name .. (default and ": ON" or ": OFF")
    Button.BackgroundColor3 = default and Color3.fromRGB(0, 150, 0) or Color3.fromRGB(60, 60, 60)
    Button.Parent = ContentFrame
    Button.MouseButton1Click:Connect(function()
        default = not default
        Button.Text = name .. (default and ": ON" or ": OFF")
        Button.BackgroundColor3 = default and Color3.fromRGB(0, 150, 0) or Color3.fromRGB(60, 60, 60)
        callback(default)
    end)
end

-- 3. CORE LOGIC
spawn(function()
    while task.wait(1) do
        if _G.AutoStats then
            local points = game.Players.LocalPlayer.Data.StatsPoints.Value
            if points > 0 then
                game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("AddPoint", "Melee", points)
            end
        end
    end
end)

local function GetNearestTarget()
    local nearest = nil
    local dist = 45 
    local char = game.Players.LocalPlayer.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        for _, v in pairs(game:GetService("Workspace").Enemies:GetChildren()) do
            if v:FindFirstChild("HumanoidRootPart") and v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
                local d = (char.HumanoidRootPart.Position - v.HumanoidRootPart.Position).Magnitude
                if d < dist then nearest = v; dist = d end
            end
        end
    end
    return nearest
end

spawn(function()
    while task.wait() do
        if _G.AutoFarm then
            pcall(function()
                local char = game.Players.LocalPlayer.Character
                if _G.SelectWeapon ~= "" and not char:FindFirstChild(_G.SelectWeapon) then
                    local tool = game.Players.LocalPlayer.Backpack:FindFirstChild(_G.SelectWeapon)
                    if tool then char.Humanoid:EquipTool(tool) end
                end
                if _G.FastAttack and GetNearestTarget() then
                    vu:CaptureController()
                    vu:Button1Down(Vector2.new(1280, 672))
                end
                if _G.BringMobs then
                    for _, v in pairs(game:GetService("Workspace").Enemies:GetChildren()) do
                        if v.Name == MonsterName and v:FindFirstChild("HumanoidRootPart") and v.Humanoid.Health > 0 then
                            v.HumanoidRootPart.CanCollide = false
                            v.HumanoidRootPart.CFrame = char.HumanoidRootPart.CFrame * CFrame.new(0, 0, -8)
                        end
                    end
                end
            end)
        end
    end
end)

spawn(function()
    while task.wait(0.1) do
        if _G.AutoFarm then
            UpdateQuest()
            pcall(function()
                local char = game.Players.LocalPlayer.Character
                local root = char:FindFirstChild("HumanoidRootPart")
                if not root then return end

                if not game:GetService("Players").LocalPlayer.PlayerGui.Main.Quest.Visible then
                    local npc = game:GetService("Workspace").NPCs:FindFirstChild(QuestNPC)
                    if npc then
                        root.CFrame = npc.HumanoidRootPart.CFrame * CFrame.new(0, 5, 0)
                        game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("StartQuest", QuestName, 1)
                    end
                else
                    local enemy = game:GetService("Workspace").Enemies:FindFirstChild(MonsterName)
                    if enemy and enemy:FindFirstChild("HumanoidRootPart") and enemy.Humanoid.Health > 0 then
                        root.CFrame = enemy.HumanoidRootPart.CFrame * CFrame.new(0, _G.AttackDistance, 0)
                        root.CFrame = CFrame.lookAt(root.Position, enemy.HumanoidRootPart.Position)
                    else
                        for _, sp in pairs(game:GetService("Workspace").EnemySpawns:GetChildren()) do
                            if sp.Name == MonsterName then
                                root.CFrame = sp.CFrame * CFrame.new(0, 20, 0)
                                break
                            end
                        end
                    end
                end
            end)
        end
    end
end)
